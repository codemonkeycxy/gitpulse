# gitpulse Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `gitpulse`, a Node.js CLI that mines git history and generates a self-contained dark HTML dashboard with 5 analytical sections: file churn+hotspots, project momentum, contributor distribution, knowledge silos, and firefighting frequency.

**Architecture:** `src/git.js` is the only caller of `child_process.execFileSync`. It exposes `run(args, repoPath)` where `args` is an array — no shell interpolation, no injection risk. Each metric in `src/metrics/` exports `collect(repoPath, opts)` returning a plain JS object. `src/renderer/html.js` merges all collected data, reads `templates/report.html`, inlines Chart.js from `node_modules`, and injects a JSON blob that client-side JS reads to render charts.

**Tech Stack:** Node.js (CommonJS, no transpilation), Commander.js v12 (CLI), Chart.js v4 (inlined in output HTML, not CDN), `open` v10 (browser launch), Jest v29 (testing)

```js
'use strict';

const { execFileSync } = require('child_process');

const OPTS = (cwd) => ({ cwd, encoding: 'utf8', stdio: 'pipe' });

/**
 * Run a git command. Args are passed as an array — no shell interpolation.
 * @param {string[]} args  e.g. ['log', '--oneline']
 * @param {string} repoPath
 * @returns {string}
 */
function run(args, repoPath) {
  return execFileSync('git', args, OPTS(repoPath));
}

/**
 * Returns true if repoPath is inside a git repository.
 * @param {string} repoPath
 * @returns {boolean}
 */
function isGitRepo(repoPath) {
  try {
    execFileSync('git', ['rev-parse', '--git-dir'], OPTS(repoPath));
    return true;
  } catch {
    return false;
  }
}

/**
 * Returns the absolute path to the git root containing repoPath.
 * @param {string} repoPath
 * @returns {string}
 */
function getGitRoot(repoPath) {
  return run(['rev-parse', '--show-toplevel'], repoPath).trim();
}

module.exports = { run, isGitRepo, getGitRoot };
```

### Task 3: src/metrics/churn.js — file churn + bug hotspots

**Files:**
- Create: `tests/metrics/churn.test.js`
- Modify: `src/metrics/churn.js`

**Algorithm:**
1. `git log --since=<since> --name-only --format= --diff-filter=ACDMRT` → all modified files
2. Count occurrences per file → churn map
3. `git log --since=<since> -E --grep=fix|bug|patch|issue|broken -i --name-only --format=` → files in bug-fix commits
4. Count per file → bugfix map
5. Top `opts.top` (default 20) by churn: category `'danger'` if bugFixes > 0, else `'churn'`
6. Files NOT in top but with bugFixes ≥ 3 → category `'bugs'`, up to 5 appended

Note: `-E` enables extended regexp so `|` means alternation without backslash.

```js
'use strict';

const git = require('../git');

function countLines(output) {
  const counts = new Map();
  for (const line of output.split('\n')) {
    const f = line.trim();
    if (f) counts.set(f, (counts.get(f) ?? 0) + 1);
  }
  return counts;
}

/**
 * Collect file churn and bug hotspot data.
 * @param {string} repoPath
 * @param {{ since: string, top?: number }} options
 * @returns {Promise<{ files: Array<{path,changes,bugFixes,category}> }>}
 */
async function collect(repoPath, { since, top = 20 }) {
  const churnOut = git.run(
    ['log', `--since=${since}`, '--name-only', '--format=', '--diff-filter=ACDMRT'],
    repoPath
  );
  const bugOut = git.run(
    ['log', `--since=${since}`, '-E', '--grep=fix|bug|patch|issue|broken', '-i', '--name-only', '--format='],
    repoPath
  );

  const churnMap = countLines(churnOut);
  const bugMap   = countLines(bugOut);

  const sorted   = [...churnMap.entries()].sort((a, b) => b[1] - a[1]);
  const topFiles = sorted.slice(0, top);
  const topPaths = new Set(topFiles.map(([p]) => p));

  const files = topFiles.map(([path, changes]) => ({
    path,
    changes,
    bugFixes: bugMap.get(path) ?? 0,
    category: (bugMap.get(path) ?? 0) > 0 ? 'danger' : 'churn',
  }));

  const extras = [...bugMap.entries()]
    .filter(([path, count]) => !topPaths.has(path) && count >= 3)
    .sort((a, b) => b[1] - a[1])
    .slice(0, 5)
    .map(([path, bugFixes]) => ({
      path,
      changes: churnMap.get(path) ?? 0,
      bugFixes,
      category: 'bugs',
    }));

  return { files: [...files, ...extras] };
}

module.exports = { collect };
```

```js
'use strict';

const git = require('../git');

function monthRange(from, to) {
  const months = [];
  let [y, m] = from.split('-').map(Number);
  const [ty, tm] = to.split('-').map(Number);
  while (y < ty || (y === ty && m <= tm)) {
    months.push(`${y}-${String(m).padStart(2, '0')}`);
    if (++m > 12) { m = 1; y++; }
  }
  return months;
}

function calcTrend(counts) {
  if (counts.length < 6) return 'stable';
  const avg = (arr) => arr.reduce((a, b) => a + b, 0) / arr.length;
  const last3  = avg(counts.slice(-3));
  const prior3 = avg(counts.slice(-6, -3));
  if (prior3 === 0) return last3 > 0 ? 'growing' : 'stable';
  if (last3 > prior3 * 1.2) return 'growing';
  if (last3 < prior3 * 0.8) return 'declining';
  return 'stable';
}

/**
 * @param {string} repoPath
 * @param {{ since: string }} options
 * @returns {Promise<{ months: string[], counts: number[], trend: string }>}
 */
async function collect(repoPath, { since }) {
  const output = git.run(
    ['log', `--since=${since}`, '--format=%ad', '--date=format:%Y-%m', '--no-merges'],
    repoPath
  );

  const raw = output.split('\n').map(l => l.trim()).filter(Boolean);
  if (raw.length === 0) return { months: [], counts: [], trend: 'stable' };

  const freq = new Map();
  for (const m of raw) freq.set(m, (freq.get(m) ?? 0) + 1);

  const sorted = [...freq.keys()].sort();
  const months = monthRange(sorted[0], sorted[sorted.length - 1]);
  const counts = months.map(m => freq.get(m) ?? 0);

  return { months, counts, trend: calcTrend(counts) };
}

module.exports = { collect };
```

```js
'use strict';

const git = require('../git');

/**
 * @param {string} repoPath
 * @param {{ since: string }} options
 * @returns {Promise<{ authors: Array<{name,commits}>, total: number, busFactor: number }>}
 */
async function collect(repoPath, { since }) {
  const output = git.run(
    ['shortlog', '-sn', '--no-merges', `--since=${since}`],
    repoPath
  );

  const lines = output.split('\n').map(l => l.trim()).filter(Boolean);
  if (lines.length === 0) return { authors: [], total: 0, busFactor: 0 };

  const parsed = lines.map(line => {
    const tab = line.indexOf('\t');
    return {
      name:    line.slice(tab + 1).trim(),
      commits: parseInt(line.slice(0, tab).trim(), 10),
    };
  }).sort((a, b) => b.commits - a.commits);

  const total = parsed.reduce((s, a) => s + a.commits, 0);

  let cumulative = 0;
  let busFactor  = 0;
  for (const author of parsed) {
    cumulative += author.commits;
    busFactor++;
    if (cumulative / total >= 0.8) break;
  }

  const TOP = 10;
  const authors = parsed.length <= TOP
    ? parsed
    : [
        ...parsed.slice(0, TOP),
        { name: 'others', commits: parsed.slice(TOP).reduce((s, a) => s + a.commits, 0) },
      ];

  return { authors, total, busFactor };
}

module.exports = { collect };
```

```js
'use strict';

const git = require('../git');

/**
 * @param {string} repoPath
 * @param {{ since: string }} options
 * @returns {Promise<{ files: Array<{path,owner,lastTouched,commits}> }>}
 */
async function collect(repoPath, { since }) {
  const listOut = git.run(
    ['log', `--since=${since}`, '--name-only', '--format=', '--diff-filter=ACDMRT'],
    repoPath
  );

  const uniqueFiles = [
    ...new Set(listOut.split('\n').map(l => l.trim()).filter(Boolean)),
  ];

  if (uniqueFiles.length === 0) return { files: [] };

  const silos = [];

  for (const file of uniqueFiles) {
    const logOut = git.run(
      ['log', `--since=${since}`, '--format=%an|%ad', '--date=format:%Y-%m-%d', '--', file],
      repoPath
    );

    const entries = logOut.split('\n').map(l => l.trim()).filter(Boolean);
    if (entries.length === 0) continue;

    const authors = new Set(entries.map(e => e.split('|')[0]));
    if (authors.size !== 1) continue;

    const dates = entries.map(e => e.split('|')[1]).sort();
    silos.push({
      path:        file,
      owner:       [...authors][0],
      lastTouched: dates[dates.length - 1],
      commits:     entries.length,
    });
  }

  silos.sort((a, b) => a.lastTouched.localeCompare(b.lastTouched));

  return { files: silos };
}

module.exports = { collect };
```

```js
'use strict';

const git = require('../git');

function monthRange(from, to) {
  const months = [];
  let [y, m] = from.split('-').map(Number);
  const [ty, tm] = to.split('-').map(Number);
  while (y < ty || (y === ty && m <= tm)) {
    months.push(`${y}-${String(m).padStart(2, '0')}`);
    if (++m > 12) { m = 1; y++; }
  }
  return months;
}

/**
 * @param {string} repoPath
 * @param {{ since: string }} options
 * @returns {Promise<{ months: string[], counts: number[], average: number, spikes: string[] }>}
 */
async function collect(repoPath, { since }) {
  const output = git.run(
    ['log', `--since=${since}`, '-E', '--grep=revert|hotfix|rollback|emergency|incident', '-i', '--format=%ad', '--date=format:%Y-%m', '--no-merges'],
    repoPath
  );

  const raw = output.split('\n').map(l => l.trim()).filter(Boolean);
  if (raw.length === 0) return { months: [], counts: [], average: 0, spikes: [] };

  const freq = new Map();
  for (const m of raw) freq.set(m, (freq.get(m) ?? 0) + 1);

  const sorted = [...freq.keys()].sort();
  const months = monthRange(sorted[0], sorted[sorted.length - 1]);
  const counts = months.map(m => freq.get(m) ?? 0);

  const average = counts.reduce((s, c) => s + c, 0) / counts.length;
  const spikes  = months.filter((m, i) => counts[i] > average * 2);

  return { months, counts, average, spikes };
}

module.exports = { collect };
```

```js
const fs   = require('fs');
```

```js
#!/usr/bin/env node
'use strict';

const path             = require('path');
const fs               = require('fs');
const { execFileSync } = require('child_process');
const { program }      = require('commander');

const git          = require('../src/git');
const churn        = require('../src/metrics/churn');
const momentum     = require('../src/metrics/momentum');
const contributors = require('../src/metrics/contributors');
const silos        = require('../src/metrics/silos');
const firefighting = require('../src/metrics/firefighting');
const { render }   = require('../src/renderer/html');

program
  .name('gitpulse')
  .description('Mine git history and generate a visual HTML repo health report')
  .argument('[repo]', 'Path to git repository (default: current directory)')
  .option('--output <file>', 'Output HTML file path', 'gitpulse-report.html')
  .option('--no-open', 'Do not open the report in the browser')
  .option('--json', 'Output raw JSON data instead of HTML')
  .option('--since <date>', 'Analyse commits since this date (YYYY-MM-DD)')
  .option('--top <n>', 'Number of top files in churn section', '20')
  .parse();

const opts = program.opts();

async function main() {
  // Verify git is installed
  try {
    execFileSync('git', ['--version'], { stdio: 'pipe' });
  } catch {
    console.error('✖ git is not installed or not in PATH.');
    console.error('  Install git: https://git-scm.com/downloads');
    process.exit(1);
  }

  const inputPath = program.args[0] ? path.resolve(program.args[0]) : process.cwd();

  if (!git.isGitRepo(inputPath)) {
    console.error(`✖ Not a git repository: ${inputPath}`);
    process.exit(1);
  }

  const repoPath = git.getGitRoot(inputPath);
  const repoName = path.basename(repoPath);

  const since = opts.since ?? (() => {
    const d = new Date();
    d.setFullYear(d.getFullYear() - 1);
    return d.toISOString().split('T')[0];
  })();

  const collectOpts = { since, top: parseInt(opts.top, 10) };

  console.log(`⚡ Analysing ${repoName} since ${since}…`);

  let results;
  try {
    results = await Promise.all([
      churn.collect(repoPath, collectOpts),
      momentum.collect(repoPath, collectOpts),
      contributors.collect(repoPath, collectOpts),
      silos.collect(repoPath, collectOpts),
      firefighting.collect(repoPath, collectOpts),
    ]);
  } catch (err) {
    console.error(`✖ Failed to collect metrics: ${err.message}`);
    process.exit(1);
  }

  const [churnData, momentumData, contributorsData, silosData, firefightingData] = results;

  const data = {
    meta: {
      repoName,
      repoPath,
      since,
      sinceLabel: '12 months',
      totalCommits: contributorsData.total,
      generatedAt:  new Date().toISOString(),
    },
    churn:        churnData,
    momentum:     momentumData,
    contributors: contributorsData,
    silos:        silosData,
    firefighting: firefightingData,
  };

  if (opts.json) {
    process.stdout.write(JSON.stringify(data, null, 2) + '\n');
    return;
  }

  let html;
  try {
    html = render(data);
  } catch (err) {
    console.error(`✖ Failed to render report: ${err.message}`);
    process.exit(1);
  }

  const outPath = path.resolve(opts.output);
  try {
    fs.writeFileSync(outPath, html, 'utf8');
  } catch (err) {
    console.error(`✖ Cannot write to ${outPath}: ${err.message}`);
    console.error(`  Try: --output /tmp/gitpulse-report.html`);
    process.exit(1);
  }

  const shouldOpen = opts.open !== false;
  console.log(`✔ Report saved to ${outPath}${shouldOpen ? ' (opening in browser…)' : ''}`);

  if (shouldOpen) {
    const { default: open } = await import('open');
    await open(outPath);
  }
}

main().catch(err => {
  console.error(`✖ ${err.message}`);
  process.exit(1);
});
```

````markdown
# gitpulse

> Mine git history. Understand any codebase before reading a single file.

`gitpulse` analyses a git repository and generates a self-contained HTML health report — a dark analytics dashboard covering the five things that matter most when approaching an unfamiliar codebase.

## Usage

```bash
# Run from inside any repo
npx gitpulse

# Or point at a path
npx gitpulse ./path/to/repo
```

## Report sections

| Section | What it shows |
|---|---|
| **File Churn + Bug Hotspots** | Top changed files, color-coded by risk (churn + bugs = danger zone) |
| **Project Momentum** | Monthly commit trend — growing, declining, or stable |
| **Contributor Distribution** | Who owns what, and your bus factor risk |
| **Knowledge Silos** | Files only one person has touched in the analysis window |
| **Firefighting Frequency** | Reverts, hotfixes, and rollback frequency with spike detection |

## Options

| Flag | Default | Description |
|---|---|---|
| `--output <file>` | `gitpulse-report.html` | Output file path |
| `--since <date>` | 1 year ago | Analyse commits since this date (YYYY-MM-DD) |
| `--top <n>` | `20` | Number of files in the churn section |
| `--no-open` | — | Don't auto-open the report in the browser |
| `--json` | — | Output raw collected data as JSON |

MIT
````
